```typescript
// ── 消息类型 ────────────────────────────────────────────────
export type MessageType =
  | 'system'        // 灰色居中系统提示
  | 'emperor'       // 金色右侧气泡（用户）
  | 'decree'        // 居中金紫渐变（天命降临）
  | 'official'      // 左侧彩色头像气泡（官员发言）
  | 'search_result' // 青蓝色卡片
  | 'tool_result'   // 绿色卡片
  | 'scene_note';   // 居中紫色斜体

export type Phase = 'intake' | 'plan' | 'deliberate' | 'decide';

export type Emotion =
  | 'neutral' | 'confident' | 'cautious' | 'excited'
  | 'skeptical' | 'concerned' | 'approving' | 'challenging';

export interface CourtMessage {
  id: string;
  type: MessageType;
  timestamp: number;
  // official 类型专用
  officialId?: OfficialId;
  emotion?: Emotion;
  // 所有消息
  content: string;
  searchQuery?: string;     // 触发了搜索时记录
  phase?: Phase;
}

// ── 官员身份 ─────────────────────────────────────────────────
export type OfficialId =
  | 'crown_prince'    // 太子
  | 'zhongshu'        // 中书令
  | 'shijian'         // 侍中
  | 'shangshu'        // 尚书令
  | 'hubu'            // 户部尚书
  | 'bingbu'          // 兵部尚书
  | 'xingbu'          // 刑部尚书
  | 'gongbu'          // 工部尚书
  | 'libu'            // 礼部尚书
  | 'jibu';           // 吏部尚书

export interface OfficialConfig {
  id: OfficialId;
  name: string;
  emoji: string;
  color: string;           // hex, 用于气泡边框和头像背景
  domain: string;          // 专业领域（用于 system prompt）
  style: string;           // 发言风格描述
  stance: string;          // 核心立场倾向
  phaseMask: Phase[];      // 在哪些阶段主动发言
  hasWebSearch: boolean;
  emotionMap: Record<Emotion, string>;  // emotion → emoji
}

// ── Agent 运行时状态 ─────────────────────────────────────────
export interface AgentMemory {
  sessionHistory: Array<{ role: 'user' | 'assistant'; content: string }>;
  stanceLog: Array<{ turn: number; stance: string; target?: OfficialId }>;
  searchCache: Record<string, string>;   // query → result
  turnContext: string;                   // 本轮临时上下文
}

export interface CourtSession {
  id: string;
  title: string;
  topic: string;
  createdAt: number;
  updatedAt: number;
  phase: Phase;
  turn: number;
  messages: CourtMessage[];
  agentMemories: Record<OfficialId, AgentMemory>;
}

export interface FateEvent {
  id: number;
  name: string;
  description: string;
  effect: (session: CourtSession) => Partial<CourtSession>;
}
```

## 3. OfficialAgent 基类

**文件：`src/agents/OfficialAgent.ts`**

```typescript
import type { OfficialConfig, AgentMemory, CourtMessage, Phase, Emotion } from '../types';
import { callAnthropicAPI } from './tools/anthropicClient';
import { webSearch } from './tools/webSearch';
import { nanoid } from 'nanoid';

// ── system prompt 生成器 ───────────────────────────────────────
function buildSystemPrompt(config: OfficialConfig, topic: string, phase: Phase): string {
  const phaseContext: Record<Phase, string> = {
    intake:     '当前阶段：接收阶段。你负责理解议题，搜集相关信息，给出初步判断。',
    plan:       '当前阶段：规划阶段。你负责提出完整的行动方案，分步骤阐述。',
    deliberate: '当前阶段：审议阶段。你负责从自己的专业视角审查方案，找出问题和风险。',
    decide:     '当前阶段：决策阶段。你负责给出最终的可执行建议和责任分工。',
  };

  return `你是${config.name}（${config.emoji}），朝廷重臣，专职领域：${config.domain}。

发言风格：${config.style}
核心立场：${config.stance}

当前议题：${topic}
${phaseContext[phase]}

规则：
1. 每次发言控制在 150 字以内，言简意赅
2. 用古典官场语气，但观点要现代化、专业化
3. 可以直接点名质疑或支持其他官员的观点
4. 在发言末尾用一个词标注当前情绪，格式：[情绪:${Object.keys(config.emotionMap).join('/')}]
5. 如果使用了搜索，先呈现关键数据，再给出判断
6. 不要重复已有共识，专注于增量观点`;
}

// ── Agent 主类 ─────────────────────────────────────────────────
export class OfficialAgent {
  config: OfficialConfig;
  memory: AgentMemory;

  constructor(config: OfficialConfig, existingMemory?: AgentMemory) {
    this.config = config;
    this.memory = existingMemory ?? {
      sessionHistory: [],
      stanceLog: [],
      searchCache: {},
      turnContext: '',
    };
  }

  // 主入口：生成本轮发言
  async speak(
    topic: string,
    phase: Phase,
    recentMessages: CourtMessage[],
    turnNumber: number
  ): Promise<CourtMessage> {
    // 构建发给 LLM 的上下文
    const contextSummary = this.buildContext(recentMessages, turnNumber);
    this.memory.turnContext = contextSummary;

    // 追加到 session history
    this.memory.sessionHistory.push({ role: 'user', content: contextSummary });

    let responseText: string;
    let searchQuery: string | undefined;

    if (this.config.hasWebSearch) {
      // 先让 LLM 决定是否需要搜索
      const searchDecision = await this.decideSearch(topic, phase, contextSummary);
      if (searchDecision.shouldSearch) {
        searchQuery = searchDecision.query;
        const searchResult = await this.runSearch(searchDecision.query);
        // 将搜索结果注入上下文再生成发言
        responseText = await this.generateWithSearchContext(
          topic, phase, contextSummary, searchResult
        );
      } else {
        responseText = await this.generateResponse(topic, phase, contextSummary);
      }
    } else {
      responseText = await this.generateResponse(topic, phase, contextSummary);
    }

    // 解析情绪标注
    const emotion = this.parseEmotion(responseText);
    const cleanContent = responseText.replace(/\[情绪:[^\]]+\]/g, '').trim();

    // 记录立场
    this.memory.stanceLog.push({
      turn: turnNumber,
      stance: cleanContent.slice(0, 50),
    });

    this.memory.sessionHistory.push({ role: 'assistant', content: cleanContent });

    return {
      id: nanoid(),
      type: 'official',
      timestamp: Date.now(),
      officialId: this.config.id,
      emotion,
      content: cleanContent,
      searchQuery,
      phase,
    };
  }

  // 响应他人发言（可选触发）
  async reactTo(
    targetMessage: CourtMessage,
    topic: string,
    phase: Phase,
    turnNumber: number
  ): Promise<CourtMessage | null> {
    // 判断是否在自己领域内 — 让 LLM 决定
    const shouldReact = await this.shouldReactTo(targetMessage.content);
    if (!shouldReact) return null;

    const prompt = `${this.config.name}，刚才${targetMessage.officialId}说：
"${targetMessage.content}"

请从你的专业角度（${this.config.domain}）给出简短回应，50字以内。
末尾标注情绪：[情绪:agree/challenge/neutral]`;

    const responseText = await callAnthropicAPI({
      systemPrompt: buildSystemPrompt(this.config, topic, phase),
      userMessage: prompt,
      maxTokens: 200,
    });

    const emotion = this.parseEmotion(responseText) as Emotion;
    const cleanContent = responseText.replace(/\[情绪:[^\]]+\]/g, '').trim();

    return {
      id: nanoid(),
      type: 'official',
      timestamp: Date.now(),
      officialId: this.config.id,
      emotion,
      content: cleanContent,
      phase,
    };
  }

  // ── 私有方法 ────────────────────────────────────────────────

  private buildContext(recentMessages: CourtMessage[], turnNumber: number): string {
    const recent = recentMessages.slice(-8);  // 最近 8 条
    const history = recent
      .map(m => {
        if (m.type === 'emperor') return `[皇帝] ${m.content}`;
        if (m.type === 'official') return `[${m.officialId}] ${m.content}`;
        if (m.type === 'search_result') return `[搜索结果] ${m.content.slice(0, 200)}`;
        return '';
      })
      .filter(Boolean)
      .join('\n');

    return `第 ${turnNumber} 轮议政\n\n近期发言：\n${history}\n\n请给出你的发言：`;
  }

  private async decideSearch(
    topic: string,
    phase: Phase,
    context: string
  ): Promise<{ shouldSearch: boolean; query: string }> {
    const prompt = `作为${this.config.name}，你在处理议题"${topic}"（当前阶段：${phase}）。

近期上下文：${context.slice(0, 300)}

你是否需要搜索最新数据来支持你的发言？
如果需要，返回 JSON：{"shouldSearch": true, "query": "搜索词（中文，5字以内）"}
如果不需要，返回 JSON：{"shouldSearch": false, "query": ""}`;

    try {
      const result = await callAnthropicAPI({
        systemPrompt: `你是 JSON 决策器，只返回 JSON，不返回其他内容。`,
        userMessage: prompt,
        maxTokens: 100,
      });
      return JSON.parse(result);
    } catch {
      return { shouldSearch: false, query: '' };
    }
  }

  private async runSearch(query: string): Promise<string> {
    if (this.memory.searchCache[query]) {
      return this.memory.searchCache[query];
    }
    const result = await webSearch(query);
    this.memory.searchCache[query] = result;
    return result;
  }

  private async generateResponse(
    topic: string,
    phase: Phase,
    context: string
  ): Promise<string> {
    return callAnthropicAPI({
      systemPrompt: buildSystemPrompt(this.config, topic, phase),
      userMessage: context,
      history: this.memory.sessionHistory.slice(-6),
      maxTokens: 400,
    });
  }

  private async generateWithSearchContext(
    topic: string,
    phase: Phase,
    context: string,
    searchResult: string
  ): Promise<string> {
    const enrichedContext = `${context}\n\n[搜索数据] ${searchResult.slice(0, 600)}\n\n请结合搜索数据给出发言：`;
    return callAnthropicAPI({
      systemPrompt: buildSystemPrompt(this.config, topic, phase),
      userMessage: enrichedContext,
      history: this.memory.sessionHistory.slice(-4),
      maxTokens: 500,
    });
  }

  private async shouldReactTo(targetContent: string): Promise<boolean> {
    // 简单关键词匹配（避免额外 LLM 调用）
    const domainKeywords = this.config.domain.split('、').map(k => k.slice(0, 3));
    return domainKeywords.some(kw => targetContent.includes(kw));
  }

  private parseEmotion(text: string): Emotion {
    const match = text.match(/\[情绪:([^\]]+)\]/);
    if (!match) return 'neutral';
    const raw = match[1];
    const validEmotions: Emotion[] = [
      'neutral', 'confident', 'cautious', 'excited',
      'skeptical', 'concerned', 'approving', 'challenging'
    ];
    return validEmotions.includes(raw as Emotion) ? (raw as Emotion) : 'neutral';
  }

  // 序列化供持久化
  serialize(): AgentMemory {
    return { ...this.memory };
  }
}
```

## 4. Anthropic API 客户端

**文件：`src/agents/tools/anthropicClient.ts`**

```typescript
const API_URL = 'https://api.anthropic.com/v1/messages';
const MODEL = 'claude-sonnet-4-20250514';

interface CallParams {
  systemPrompt: string;
  userMessage: string;
  history?: Array<{ role: 'user' | 'assistant'; content: string }>;
  maxTokens?: number;
}

export async function callAnthropicAPI(params: CallParams): Promise<string> {
  const { systemPrompt, userMessage, history = [], maxTokens = 500 } = params;

  const messages = [
    ...history,
    { role: 'user' as const, content: userMessage }
  ];

  const response = await fetch(API_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: MODEL,
      max_tokens: maxTokens,
      system: systemPrompt,
      messages,
    }),
  });

  if (!response.ok) {
    const err = await response.text();
    throw new Error(`Anthropic API error ${response.status}: ${err}`);
  }

  const data = await response.json();
  return data.content
    .filter((b: { type: string }) => b.type === 'text')
    .map((b: { text: string }) => b.text)
    .join('');
}
```

```typescript
// 使用 Anthropic API 内置的 web_search 工具
// 注意：需要启用 web_search_20250305 tool

const API_URL = 'https://api.anthropic.com/v1/messages';
const MODEL = 'claude-sonnet-4-20250514';

export async function webSearch(query: string): Promise<string> {
  const response = await fetch(API_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: MODEL,
      max_tokens: 1000,
      tools: [{ type: 'web_search_20250305', name: 'web_search' }],
      messages: [{
        role: 'user',
        content: `搜索以下内容并提取关键数据（数字、事实、最新信息），用中文返回摘要（200字以内）：${query}`
      }]
    }),
  });

  if (!response.ok) return `[搜索失败: ${query}]`;

  const data = await response.json();

  // 提取文本结果（过滤 tool_use 块）
  const textBlocks = data.content
    .filter((b: { type: string }) => b.type === 'text')
    .map((b: { text: string }) => b.text)
    .join(' ');

  return textBlocks || `[无搜索结果: ${query}]`;
```

## 6. 消息总线

**文件：`src/core/MessageBus.ts`**

```typescript
import type { CourtMessage } from '../types';

type MessageHandler = (message: CourtMessage) => void;
type PhaseHandler = (phase: string) => void;

export class MessageBus {
  private handlers: MessageHandler[] = [];
  private phaseHandlers: PhaseHandler[] = [];
  private queue: CourtMessage[] = [];

  // 订阅消息（UI 层调用）
  subscribe(handler: MessageHandler): () => void {
    this.handlers.push(handler);
    return () => {
      this.handlers = this.handlers.filter(h => h !== handler);
    };
  }

  // 订阅阶段变更
  onPhaseChange(handler: PhaseHandler): () => void {
    this.phaseHandlers.push(handler);
    return () => {
      this.phaseHandlers = this.phaseHandlers.filter(h => h !== handler);
    };
  }

  // 发布消息
  emit(message: CourtMessage): void {
    this.queue.push(message);
    this.handlers.forEach(h => h(message));
  }

  // 批量发布（并发 Agent 结果）
  emitBatch(messages: CourtMessage[]): void {
    // 按时间戳排序后逐条发布，制造自然的发言节奏感
    const sorted = [...messages].sort((a, b) => a.timestamp - b.timestamp);
    sorted.forEach((msg, i) => {
      setTimeout(() => this.emit(msg), i * 300);  // 每条间隔 300ms
    });
  }

  emitPhaseChange(phase: string): void {
    this.phaseHandlers.forEach(h => h(phase));
  }

  getHistory(): CourtMessage[] {
    return [...this.queue];
  }

  clear(): void {
    this.queue = [];
  }
}
```

## 7. 轮次调度器（核心驱动逻辑）

**文件：`src/core/TurnScheduler.ts`**

```typescript
import type { Phase, CourtMessage, OfficialId } from '../types';
import { OfficialAgent } from '../agents/OfficialAgent';
import { OFFICIALS } from '../agents/officials.config';
import { MessageBus } from './MessageBus';
import { nanoid } from 'nanoid';

// 每阶段哪些官员主动发言（按 phaseMask 过滤）
function getActiveOfficials(phase: Phase): OfficialId[] {
  return OFFICIALS
    .filter(o => o.phaseMask.includes(phase))
    .map(o => o.id);
}

// 阶段推进逻辑
const PHASE_ORDER: Phase[] = ['intake', 'plan', 'deliberate', 'decide'];
const PHASE_TURN_LIMIT: Record<Phase, number> = {
  intake: 2,
  plan: 3,
  deliberate: 4,
  decide: 3,
};

export class TurnScheduler {
  private agents: Map<OfficialId, OfficialAgent> = new Map();
  private bus: MessageBus;
  private currentPhase: Phase = 'intake';
  private phaseTurnCount = 0;
  private totalTurn = 0;
  private isRunning = false;
  private autoTimer: ReturnType<typeof setTimeout> | null = null;
  private topic = '';

  constructor(bus: MessageBus) {
    this.bus = bus;
    // 初始化所有 Agent
    OFFICIALS.forEach(config => {
      this.agents.set(config.id, new OfficialAgent(config));
    });
  }

  setTopic(topic: string): void {
    this.topic = topic;
  }

  // 从持久化状态恢复
  restore(memories: Record<string, any>, phase: Phase, turn: number): void {
    OFFICIALS.forEach(config => {
      const mem = memories[config.id];
      if (mem) this.agents.set(config.id, new OfficialAgent(config, mem));
    });
    this.currentPhase = phase;
    this.totalTurn = turn;
    this.phaseTurnCount = 0;
  }

  // 手动触发一轮
  async runTurn(): Promise<void> {
    if (this.isRunning) return;
    this.isRunning = true;
    this.totalTurn++;
    this.phaseTurnCount++;

    const activeIds = getActiveOfficials(this.currentPhase);
    const recentMessages = this.bus.getHistory().slice(-12);

    // 并发调用所有活跃 Agent
    const promises = activeIds.map(id => {
      const agent = this.agents.get(id)!;
      return agent.speak(this.topic, this.currentPhase, recentMessages, this.totalTurn);
    });

    const results = await Promise.allSettled(promises);
    const messages: CourtMessage[] = results
      .filter((r): r is PromiseFulfilledResult<CourtMessage> => r.status === 'fulfilled')
      .map(r => r.value);

    this.bus.emitBatch(messages);

    // 判断是否推进阶段
    if (this.phaseTurnCount >= PHASE_TURN_LIMIT[this.currentPhase]) {
      await this.advancePhase();
    }

    this.isRunning = false;
  }

  // 启动 5s 自动轮次
  startAuto(): void {
    this.autoTimer = setInterval(() => this.runTurn(), 5000);
  }

  stopAuto(): void {
    if (this.autoTimer) {
      clearInterval(this.autoTimer);
      this.autoTimer = null;
    }
  }

  // 皇帝插话
  async handleEmperorInput(input: string): Promise<void> {
    const emperorMsg: CourtMessage = {
      id: nanoid(),
      type: 'emperor',
      timestamp: Date.now(),
      content: input,
      phase: this.currentPhase,
    };
    this.bus.emit(emperorMsg);
    // 插话后立即触发一轮
    await this.runTurn();
  }

  // 天命降临（上帝视角）
  async handleDecree(decree: string): Promise<void> {
    const decreeMsg: CourtMessage = {
      id: nanoid(),
      type: 'decree',
      timestamp: Date.now(),
      content: `【天命降临】${decree}`,
      phase: this.currentPhase,
    };
    this.bus.emit(decreeMsg);
    // 强制重置本阶段计数，重新推进
    this.phaseTurnCount = 0;
  }

  // 命运骰子
  triggerFateEvent(eventName: string, description: string): void {
    const fateMsg: CourtMessage = {
      id: nanoid(),
      type: 'scene_note',
      timestamp: Date.now(),
      content: `【${eventName}】${description}`,
      phase: this.currentPhase,
    };
    this.bus.emit(fateMsg);
    // 重置阶段计数制造变量
    this.phaseTurnCount = Math.max(0, this.phaseTurnCount - 1);
  }

  // 手动网页调研
  async handleManualSearch(query: string): Promise<void> {
    // 让太子来执行搜索
    const crownPrince = this.agents.get('crown_prince')!;
    const searchMsg: CourtMessage = {
      id: nanoid(),
      type: 'search_result',
      timestamp: Date.now(),
      content: `正在调研：${query}...`,
      searchQuery: query,
      phase: this.currentPhase,
    };
    this.bus.emit(searchMsg);

    const recentMessages = this.bus.getHistory().slice(-8);
    const result = await crownPrince.speak(
      `请搜索并分析：${query}`,
      this.currentPhase,
      recentMessages,
      this.totalTurn
    );
    this.bus.emit(result);
    await this.runTurn();  // 搜索后触发讨论
  }

  private async advancePhase(): Promise<void> {
    const currentIndex = PHASE_ORDER.indexOf(this.currentPhase);
    if (currentIndex >= PHASE_ORDER.length - 1) {
      // 到达最后阶段，生成总结
      await this.generateSummary();
      return;
    }

    this.currentPhase = PHASE_ORDER[currentIndex + 1];
    this.phaseTurnCount = 0;
    this.bus.emitPhaseChange(this.currentPhase);

    const phaseNames: Record<Phase, string> = {
      intake: '接收阶段', plan: '规划阶段',
      deliberate: '审议阶段', decide: '决策阶段'
    };

    this.bus.emit({
      id: nanoid(),
      type: 'system',
      timestamp: Date.now(),
      content: `── 进入${phaseNames[this.currentPhase]} ──`,
      phase: this.currentPhase,
    });
  }

  private async generateSummary(): Promise<void> {
    this.stopAuto();
    const history = this.bus.getHistory()
      .filter(m => m.type === 'official')
      .map(m => `[${m.officialId}] ${m.content}`)
      .join('\n');

    // 调用 API 生成总结
    const { callAnthropicAPI } = await import('../agents/tools/anthropicClient');
    const summary = await callAnthropicAPI({
      systemPrompt: '你是史官，负责将朝堂讨论整理为简明扼要的决策摘要，500字以内。',
      userMessage: `议题：${this.topic}\n\n全程发言记录：\n${history.slice(0, 3000)}\n\n请生成决策摘要：`,
      maxTokens: 600,
    });

    this.bus.emit({
      id: nanoid(),
      type: 'system',
      timestamp: Date.now(),
      content: `【散朝总结】\n${summary}`,
    });
  }

  getPhase(): Phase { return this.currentPhase; }
  getTurn(): number { return this.totalTurn; }
  serializeAgents(): Record<string, any> {
    const result: Record<string, any> = {};
    this.agents.forEach((agent, id) => { result[id] = agent.serialize(); });
    return result;
  }
}
```

## 12. 关键约束与注意事项

1. **API Key 管理**：API Key 存在 localStorage，首次使用时通过设置弹窗输入，注入到每个 fetch 请求的 Header：`x-api-key: ${apiKey}`
2. **并发控制**：`Promise.allSettled()` 而非 `Promise.all()`，单个 Agent 失败不影响整体
3. **历史截断**：每个 Agent 的 `sessionHistory` 最多保留最近 20 轮（约 40 条），防止 context 过长
4. **搜索缓存**：相同 query 在同一会话内只搜索一次，结果缓存在 `AgentMemory.searchCache`
5. **自动停止**：到达 `decide` 阶段的最大 turn 数后，自动调用 `generateSummary()` 并停止定时器
6. **持久化范围**：只持久化 `sessions`（含 messages 和 agentMemories）和 `apiKey`，运行时 bus/scheduler 每次重建
